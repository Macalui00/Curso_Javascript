<!--Todo lo que tenemos que haces escribir <script></script>-->
<!--Lo que vamos a hacer ahora es que simplemente imprima por consola del desarrollador de javascript algo
de texto.-->   

<!--
<script>
    console.log('Hola mundo'); 
</script> -->

inspeccionar, y en la seccion consola nos aparecera el Hola mundo-->

<!--Ahora con codigo javascript en un archivo aparte-->
<script src="HolaMundo.js"></script>
